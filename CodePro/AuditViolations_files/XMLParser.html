122       for (int n = childNodes.getLength() - 1; n >= 0; n--) {
123          Node child = childNodes.item(n);
124          short nodeType = child.getNodeType();
125          if (nodeType == Node.ELEMENT_NODE) {
126             clean(child);
127          } else if (nodeType == Node.TEXT_NODE) {
128             String trimmedNodeVal = child.getNodeValue().trim();
129             if (trimmedNodeVal.length() == 0) {
130                node.removeChild(child);
131             } else {
132                child.setNodeValue(trimmedNodeVal);
133             }
134          } else if (nodeType == Node.COMMENT_NODE) {
135             node.removeChild(child);
